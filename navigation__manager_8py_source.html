58 def load_waypoints(self) -> dict:
59 """Load waypoints from YAML file.
60
61 Returns:
62 Dictionary of waypoints with name as key and coordinates as value.
63 Empty dict if file not found or parsing fails.
64 """
65 if not os.path.exists(self.waypoints_path):
66 return {}
67 try:
68 with open(self.waypoints_path, 'r') as f:
69 data = yaml.safe_load(f)
70 return data.get('waypoints', {})
71 except Exception as e:
72 self.get_logger().error(f"Error reading waypoints file: {e}")
73 return {}

75 def euler_to_quaternion(self, yaw: float) -> dict:
76 """Convert Euler yaw angle to quaternion representation.
77
78 Args:
79 yaw: Yaw angle in radians
80
81 Returns:
82 Dictionary with 'z' and 'w' quaternion components

84 return {
85 'z': math.sin(yaw / 2),

89 def run(self):
90 """Execute waypoint navigation in sequential or random order.
91
92 Navigates through all loaded waypoints, monitoring progress and
93 handling navigation results (success, cancel, failure).
94 """
95 wp_names = list(self.waypoints.keys())
96
97 if self.mode == 'random':
98 random.shuffle(wp_names)
99 self.get_logger().info("Mode: Random")
100 else:
101 self.get_logger().info("Mode: Sequential")
102
103 for name in wp_names:
104 wp = self.waypoints[name]
105 self.get_logger().info(f"Navigating to: {name} (x: {wp['x']}, y: {wp['y']})")
106
107 goal_pose = PoseStamped()
108 goal_pose.header.frame_id = 'map'
109 goal_pose.header.stamp = self.navigator.get_clock().now().to_msg()
110 goal_pose.pose.position.x = float(wp['x'])
111 goal_pose.pose.position.y = float(wp['y'])
112
113 q = self.euler_to_quaternion(float(wp['theta']))
114 goal_pose.pose.orientation.z = q['z']
115 goal_pose.pose.orientation.w = q['w']
116
117 # Send robot to goal pose
118 self.navigator.goToPose(goal_pose)
119
120 # Monitor navigation progress
121 while not self.navigator.isTaskComplete():
122 feedback = self.navigator.getFeedback()
123 if feedback:
124 pass
125 time.sleep(1.0)
126
127 # Check navigation result
128 result = self.navigator.getResult()
129 if result == TaskResult.SUCCEEDED:
130 self.get_logger().info(f"Reached {name}!")
131 # self.speak(f"He llegado a {name}") # Commented: announcement done by voice_controller if active
132 # Pause between waypoints
133 time.sleep(2.0)
134 elif result == TaskResult.CANCELED:
135 self.get_logger().warn(f"Mission canceled for {name}.")
136 break
137 elif result == TaskResult.FAILED:
138 self.get_logger().error(f"Failed to navigate to {name}.")
139 break
