Detailed Description

ROS2 node that converts LiDAR scans to 8x8 occupancy grid for Arduino visualization.

Subscribes to /scan topic and publishes grid data via serial port at ~5 Hz.

105 angle += msg.angle_increment

107

113 x = r * math.cos(angle)
114 y = r * math.sin(angle)

117 # Robot is at center between cells [3,3], [3,4], [4,3], [4,4]
118 # Grid X: 0 = front (+half_grid), 7 = back (-half_grid)
119 # Grid Y: 0 = right (-half_grid), 7 = left (+half_grid)
120
121 # Check if point is within grid bounds
122 if abs(x) < self.half_grid and abs(y) < self.half_grid:
123 # Convert to grid indices (0-7)
124 # Row: front (x=+max) -> row 0, back (x=-max) -> row 7
125 # Col: left (y=+max) -> col 0, right (y=-max) -> col 7
126 # (Inverted to correct 180° rotation)
127 row = int((x + self.half_grid) / self.cell_size)
128 col = int((self.half_grid - y) / self.cell_size)
129
130 # Clamp to valid range

132 col = max(0, min(7, col))

139 # 90° counterclockwise rotation: new[7-col][row] = old[row][col]

147 byte_val = 0

151 hex_bytes.append(f"{byte_val:02X}")
