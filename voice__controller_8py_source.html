2"""
3Voice Controller Node - Spanish voice command interface for robot navigation.
4
5This node provides complete voice control using Vosk STT (speech-to-text) and
6Piper TTS (text-to-speech). It supports navigation to predefined locations,
7coordinate-based navigation, patrol commands, map saving, and stop commands.
8
9Voice Commands (in Spanish):
10 - "ve a [location]" - Navigate to predefined location
11 - "ve a [x] [y]" - Navigate to coordinates
12 - "patrulla" - Start sequential waypoint patrol
13 - "patrulla aleatoria" - Start random waypoint patrol
14 - "para" / "stop" - Stop all navigation and patrol
15 - "guardar mapa" - Save current SLAM map
16 - "opciones" - List available commands
17
18Predefined Locations:
19 sofá, cocina, almacén, principal, baño, secundaria, inicio
20

22 - Python virtual environment at PROJECT_ROOT/.venv
23 - Vosk model: vosk-model-small-es-0.42
24 - Piper TTS: es_ES-carlfm-x_low.onnx
25 - sounddevice, vosk packages
26 - Microphone and speakers
27
28Usage:
29 # Launch with dedicated script (activates venv)
30 ./scripts/launch_voice.sh
31
32 # Or directly (ensure venv is activated)
33 ros2 run home_robot voice_controller
34
35Architecture:
36 - STT thread: Continuous audio capture and recognition
37 - TTS thread: Non-blocking speech synthesis

167 def say(self, text: str) -> None:
168 """
169 Synthesize and play Spanish text using Piper TTS.
170
171 Non-blocking method that runs TTS in a separate thread. Sets is_speaking
172 flag to prevent audio input interference during speech output.
173
174 Args:
175 text: Spanish text to synthesize and speak
176 """
177 def _speak():
178 self.is_speaking = True
179 self.get_logger().info(f"Diciendo: {text}")
180 try:
181 # Verifica binario y permiso antes de hablar
182 if not os.path.isfile(self.piper_path):
183 self.get_logger().error(f"Piper no encontrado en {self.piper_path}")
184 return
185 if not os.access(self.piper_path, os.X_OK):
186 self.get_logger().error("Piper no es ejecutable; ejecuta chmod +x sobre el binario")
187 return
188
189 # Generate audio with Piper and play with aplay
190 env = os.environ.copy()
191 env.setdefault("OMP_NUM_THREADS", "1")
192 env.setdefault("OMP_WAIT_POLICY", "PASSIVE")
193
194 run_tts = subprocess.run(
195 [self.piper_path, "--model", self.piper_model, "--config", self.piper_config, "--output_file", "/tmp/tts.wav"],
196 input=text.encode('utf-8'),
197 stdout=subprocess.DEVNULL,
198 stderr=subprocess.PIPE,
199 env=env,
200 check=False
201 )
202
203 if run_tts.returncode != 0:
204 err_text = run_tts.stderr.decode('utf-8', errors='ignore') if run_tts.stderr else ''
205 self.get_logger().error(f"Piper falló (código={run_tts.returncode}): {err_text[:200]}")
206 return
207
208 play = subprocess.run(["aplay", "/tmp/tts.wav"], stdout=subprocess.DEVNULL, stderr=subprocess.PIPE, env=env, check=False)
209 if play.returncode != 0:
210 err_text = play.stderr.decode('utf-8', errors='ignore') if play.stderr else ''
211 self.get_logger().error(f"aplay falló (código={play.returncode}): {err_text[:200]}")
212 return
213 except Exception as e:
214 self.get_logger().error(f"Error en TTS: {e}")
215 finally:
216 time.sleep(0.5)
217 self.is_speaking = False
218
219 threading.Thread(target=_speak).start()

258 """
259 Continuous voice recognition loop.
260
261 Runs in separate thread, processing queued audio data through Vosk
262 recognizer and dispatching recognized commands.
263 """
264 try:
265 with sd.RawInputStream(samplerate=16000, blocksize=8000, dtype='int16',
266 channels=1, callback=self.audio_callback):

291 def process_command(self, text: str) -> None:
292 """
293 Process recognized voice command and execute corresponding action.
294
295 Handles stop commands, location navigation, coordinate navigation,
296 patrol commands, map saving, and options listing.
297
298 Args:
299 text: Recognized text from voice input (lowercase Spanish)
300 """
301 # Detect stop command (with various recognition variations)
302 if any(w in text for w in ["para", "pará", "parar", "stop", "detente", "detén", "alto"]):
303 self.get_logger().info("🛑 Comando de parada detectado")
304
305 # STEP 0: Send ROS-native stop signal first (most robust)
306 stop_msg = String()
307 stop_msg.data = "STOP"
308 self.stop_pub.publish(stop_msg)
309 self.get_logger().info("Señal de parada enviada por tópico /stop_patrol")
310
311 # STEP 1: Send SIGTERM to execute_patrol...
312 try:
313 # First SIGTERM to allow cleanup opportunity
314 result = subprocess.run(
315 ["pkill", "-TERM", "-f", "execute_patrol"],
316 capture_output=True,
317 timeout=1
318 )
319 self.get_logger().info(f"SIGTERM enviado (código: {result.returncode})")
320
321 # Wait 0.5s for signal processing
322 time.sleep(0.5)
323 except Exception as e:
324 self.get_logger().warn(f"Error enviando SIGTERM: {e}")
325
326 # STEP 2: If registered process still alive, kill with SIGTERM
327 if self.patrol_process:
328 try:
329 if self.patrol_process.poll() is None: # If still alive
330 self.get_logger().info(f"Terminando PID registrado con SIGTERM: {self.patrol_process.pid}")
331 # Send SIGTERM to session group
332 os.killpg(os.getpgid(self.patrol_process.pid), signal.SIGTERM)
333 time.sleep(0.3)
334 except:
335 pass
336
337 # STEP 3: If STILL alive, use SIGKILL
338 try:
339 result = subprocess.run(
340 ["pkill", "-9", "-f", "execute_patrol"],
341 capture_output=True,
342 timeout=1
343 )
344 if result.returncode == 0:
345 self.get_logger().info("SIGKILL enviado como respaldo")
346 except:
347 pass
348
349 # STEP 4: Clean process reference
350 if self.patrol_process:
351 try:
352 self.patrol_process.wait(timeout=0.2)
353 except:
354 pass
355 self.patrol_process = None
356
357 # STEP 5: Mark local navigation as inactive (the thread will handle cancellation)
358 self.is_navigating = False
359
360 # STEP 6: Clear visual path
361 self.cancel_all_navigation()
362
363 # STEP 7: Publish zero velocity MULTIPLE TIMES (immediate physical stop)
364 self.get_logger().info("Publicando cmd_vel cero (frenado inmediato)...")
365 stop_msg = Twist()
366 for _ in range(20):
367 self.cmd_vel_pub.publish(stop_msg)
368 time.sleep(0.01)
369
370 self.get_logger().info("✓ Robot detenido completamente")
371 self.say("Entendido, me detengo.")
372 return
373
374 if "opciones" in text or "opción" in text:
375 # List main destinations (without synonyms)
376 destinos = ["sofá", "cocina", "almacén", "principal", "baño", "secundaria", "inicio"]
377 mensaje = "Puedo ir a los siguientes lugares: " + ", ".join(destinos) + ". También puedo ir a unas coordenadas específicas, hacer una patrulla, una patrulla aleatoria, o guardar el mapa."
378 self.say(mensaje)
379 return
380
381 # Detect map save command
382 if "guardar" in text and "mapa" in text:
383 self.say("Guardando el mapa.")
384 self.save_map()
385 return
386
387 # Detect patrol command
388 if "patrulla" in text:
389 # Cancel active navigation before starting patrol
390 if self.is_navigating:
391 self.get_logger().info("Cancelando navegación para iniciar patrulla...")
392 self.nav.cancelTask()
393 self.is_navigating = False
394 time.sleep(0.3)
395
396 if "aleatorio" in text or "aleatoria" in text or "random" in text:
397 self.say("Iniciando patrulla aleatoria.")
398 self.launch_patrol(random=True)
399 else:
400 self.say("Iniciando patrulla secuencial.")
401 self.launch_patrol(random=False)
402 return
403
404 # Detect coordinate commands (e.g., "ve a uno uno", "ir a dos tres")
405 coords = self.extract_coordinates(text)
406 if coords:
407 x, y = coords
408 self.say(f"Entendido. Voy a las coordenadas {x}, {y}")
409 self.go_to([x, y], f"coordenadas {x}, {y}")
410 return
411
412 for key, coords in self.locations.items():
413 if key in text:
414 self.say(f"Entendido. Voy a {key}")
415 self.go_to(coords, key)
416 return

418 def extract_coordinates(self, text: str):
419 """
420 Extract numeric coordinates from Spanish voice command.
421
422 Converts Spanish number words to floats for navigation.
423 Examples:
424 "ve a uno dos" -> (1.0, 2.0)
425 "ir a menos tres cinco" -> (-3.0, 5.0)
426
427 Args:
428 text: Recognized Spanish voice command
429
430 Returns:
431 tuple: (x, y) coordinates as floats, or None if not found
432 """
433 import re
434
435 # Look for patterns like "ve a", "ir a", "coordenadas"
436 if not any(pattern in text for pattern in ["ve a", "ir a", "coordenadas"]):
437 return None
438
439 # Extract words after command
440 words = text.split()
441 numbers = []
442 i = 0
443 sign = 1
444
445 while i < len(words):
446 word = words[i]
447
448 # Detect negative sign
449 if word == "menos":
450 sign = -1
451 i += 1
452 continue
453
454 # Convert word to number
455 if word in self.numero_map:
456 num_value = self.numero_map[word]
457 if isinstance(num_value, int):
458 numbers.append(float(sign * num_value))
459 sign = 1
460 # Detect numbers written as digits
461 elif word.isdigit() or (word.replace('-', '').replace('.', '').isdigit()):
462 try:
463 numbers.append(float(word))
464 sign = 1
465 except:
466 pass
467
468 i += 1
469
470 # Need at least 2 numbers (x, y)
471 if len(numbers) >= 2:
472 return (numbers[0], numbers[1])
473
474 return None

476 def go_to(self, coords: list, name: str) -> None:
477 """
478 Navigate to specified coordinates using Nav2.
479
480 Cancels any active navigation/patrol before starting new goal.
481 Runs navigation in separate thread to avoid blocking voice recognition.
482
483 Args:
484 coords: [x, y] coordinates in map frame
485 name: Descriptive name for TTS announcements
486 """
487 # Cancel previous navigation if exists
488 if self.is_navigating:
489 self.get_logger().info("Cancelando navegación anterior...")
490 self.nav.cancelTask()
491 time.sleep(0.5) # Wait longer for cancellation to process completely
492
493 # Cancel active patrol process if exists
494 if self.patrol_process and self.patrol_process.poll() is None:
495 self.get_logger().info(f"Terminando proceso de patrulla (PID: {self.patrol_process.pid})")
496 self.patrol_process.terminate()
497 time.sleep(0.3)
498 if self.patrol_process.poll() is None:
499 self.patrol_process.kill()
500 self.patrol_process = None
501
502 goal = PoseStamped()
503 goal.header.frame_id = 'map'
504 goal.header.stamp = self.get_clock().now().to_msg()
505 goal.pose.position.x = float(coords[0])
506 goal.pose.position.y = float(coords[1])
507 goal.pose.orientation.w = 1.0
508
509 def _nav_task():
510 self.is_navigating = True
511 try:
512 # Send goal
513 self.nav.goToPose(goal)
514
515 # Verify future exists
516 if not hasattr(self.nav, 'result_future') or self.nav.result_future is None:
517 self.get_logger().error(f"No se pudo iniciar navegación a {name}")
518 self.say("Error al iniciar la navegación.")
519 return
520
521 # Wait for navigation to complete
522 while not self.nav.isTaskComplete() and self.is_navigating:
523 time.sleep(0.1)
524
525 # If exited via manual cancellation, perform actual task cancellation here (thread-safe)
526 if not self.is_navigating:
527 self.get_logger().info("Cancelando tarea en Nav2 de forma segura...")
528 try:
529 self.nav.cancelTask()
530 except:
531 pass
532 return
533
534 # Get result
535 result = self.nav.getResult()
536
537 # Check if we are close enough even if result isn't perfect
538 # (Simulation sometimes returns FAILED or UNKNOWN if tolerances are tight)
539 if result == TaskResult.SUCCEEDED:
540 self.get_logger().info(f"✓ Navegación a {name} completada con éxito")
541 self.say(f"Ya he llegado a {name}")
542 else:
543 self.get_logger().warn(f"Navegación a {name} terminó con resultado: {result}")
544 # If we finished the loop, we are likely at the goal even if result isn't 'Succeeded'
545 self.say(f"He llegado a {name}")
546 except KeyboardInterrupt:
547 self.get_logger().info("Navegación interrumpida por el usuario")
548 self.nav.cancelTask()
549 except Exception as e:
550 self.get_logger().error(f"Error en navegación: {e}")
551 try:
552 self.say("Ocurrió un error durante la navegación.")
553 except:
554 pass
555 finally:
556 self.is_navigating = False
557
558 self.nav_thread = threading.Thread(target=_nav_task, daemon=True)
559 self.nav_thread.start()
