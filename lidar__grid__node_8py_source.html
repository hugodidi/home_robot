2"""
3LiDAR Grid Node - Discretizes /scan into 8x8 grid and sends via Serial to Arduino.
4
5This node subscribes to laser scan data, converts it to a discretized 8x8 occupancy
6grid, and transmits the grid to an Arduino via serial communication. The robot is
7assumed to be at the center of the grid.
8
9Grid Orientation:
10 - Row 0 = front of robot, Row 7 = back
11 - Col 0 = right side, Col 7 = left side
12 - Robot position: center between cells [3,3], [3,4], [4,3], [4,4]
13
14Serial Protocol:
15 - Distance message: "D:{min_dist:.2f}\n"
16 - Grid message: "M:{hex1},{hex2},...,{hex8}\n"
17 - Each hex byte represents one row (8 bits for 8 columns)
18
19Usage:
20 ros2 run home_robot lidar_grid_node --port /dev/ttyUSB0 --baud 115200 --grid 0.25
21"""

82 def scan_callback(self, msg: LaserScan) -> None:
83 """
84 Process incoming laser scan and generate occupancy grid.
85
86 Args:
87 msg: LaserScan message from /scan topic
88 """
89 # Rate limiting
90 now = self.get_clock().now()
91 elapsed = (now - self.last_send_time).nanoseconds
92 if elapsed < self.send_interval_ns:
93 return
94 self.last_send_time = now
95
96 # Initialize 8x8 grid (all zeros)
97 grid = [[0] * self.grid_size for _ in range(self.grid_size)]
98
99 min_distance = float('inf')
100 angle = msg.angle_min
101
102 for r in msg.ranges:
103 # Filter invalid ranges
104 if math.isnan(r) or math.isinf(r) or r < msg.range_min or r > msg.range_max:
105 angle += msg.angle_increment
106 continue
107
108 # Track minimum distance
109 if r < min_distance:
110 min_distance = r
111
112 # Convert polar to Cartesian (robot frame: X forward, Y left)
113 x = r * math.cos(angle)
114 y = r * math.sin(angle)
115
116 # Map to grid cell
117 # Robot is at center between cells [3,3], [3,4], [4,3], [4,4]
118 # Grid X: 0 = front (+half_grid), 7 = back (-half_grid)
119 # Grid Y: 0 = right (-half_grid), 7 = left (+half_grid)
120
121 # Check if point is within grid bounds
122 if abs(x) < self.half_grid and abs(y) < self.half_grid:
123 # Convert to grid indices (0-7)
124 # Row: front (x=+max) -> row 0, back (x=-max) -> row 7
125 # Col: left (y=+max) -> col 0, right (y=-max) -> col 7
126 # (Inverted to correct 180Â° rotation)
127 row = int((x + self.half_grid) / self.cell_size)
128 col = int((self.half_grid - y) / self.cell_size)
129
130 # Clamp to valid range
131 row = max(0, min(7, row))
132 col = max(0, min(7, col))
133
134 grid[row][col] = 1
135
136 angle += msg.angle_increment
137
138 # Convert grid to hex bytes
139 # 90Â° counterclockwise rotation: new[7-col][row] = old[row][col]
140 rotated_grid = [[0] * self.grid_size for _ in range(self.grid_size)]
141 for row in range(self.grid_size):
142 for col in range(self.grid_size):
143 rotated_grid[7 - col][row] = grid[row][col]
144
145 hex_bytes = []
146 for row in reversed(rotated_grid): # 180Â°: reverse rows
147 byte_val = 0
148 for col, val in enumerate(row):
149 if val:
150 byte_val |= (1 << (7 - col)) # 180Â°: reverse columns
151 hex_bytes.append(f"{byte_val:02X}")
152
153 # Handle case where no valid ranges were found
154 if min_distance == float('inf'):
155 min_distance = 10.0 # Default safe value
156
157 # Build messages
158 dist_msg = f"D:{min_distance:.2f}\n"
159 grid_msg = f"M:{','.join(hex_bytes)}\n"
160
161 # Log for debugging
162 self.get_logger().debug(f"Min dist: {min_distance:.2f}m | Grid: {','.join(hex_bytes)}")
163
164 # Send via serial
165 if self.serial_enabled and self.ser and self.ser.is_open:
166 try:
167 self.ser.write(dist_msg.encode())
168 self.ser.write(grid_msg.encode())
169 except serial.SerialException as e:
170 self.get_logger().warn(f"Serial error: {e}")
171 else:
172 # Print to console for testing
173 self.get_logger().info(f"[TEST] {dist_msg.strip()} | {grid_msg.strip()}")
