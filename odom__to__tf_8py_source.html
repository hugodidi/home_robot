28 def odom_callback(self, msg: Odometry) -> None:
29 """Process odometry message and publish TF transform.

31 Args:
32 msg: Odometry message containing pose and frame information.

34 t = TransformStamped()

36 t.header.frame_id = msg.header.frame_id or "odom"
37 t.child_frame_id = msg.child_frame_id or "base_footprint"
38 t.transform.translation.x = msg.pose.pose.position.x
39 t.transform.translation.y = msg.pose.pose.position.y
40 t.transform.translation.z = msg.pose.pose.position.z
41 t.transform.rotation = msg.pose.pose.orientation
