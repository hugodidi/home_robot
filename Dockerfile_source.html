1# Use an optimized image that already includes desktop and VNC for Jazzy
2FROM tiryoh/ros2-desktop-vnc:jazzy
3
4# Avoid interactive prompts
5ENV DEBIAN_FRONTEND=noninteractive
6
7# Switch to root to install system dependencies
8USER root
9
10# 1. Install hardware and voice dependencies
11RUN apt-get update && apt-get install -y \
12 alsa-utils \
13 libasound2-dev \
14 libportaudio2 \
15 python3-pip \
16 python3-venv \
17 curl \
18 unzip \
19 tar \
20 python3-serial \
21 vlc-bin \
22 # Project-specific ROS 2 dependencies
23 ros-jazzy-nav2-bringup \
24 ros-jazzy-nav2-simple-commander \
25 ros-jazzy-slam-toolbox \
26 ros-jazzy-ros-gz \
27 ros-jazzy-joint-state-publisher \
28 ros-jazzy-xacro \
29 ros-jazzy-turtlebot3-description \
30 ros-jazzy-tf2-ros \
31 ros-jazzy-robot-state-publisher \
32 ros-jazzy-rmw-cyclonedds-cpp \
33 libasound2-plugins \
34 pulseaudio-utils \
35 && rm -rf /var/lib/apt/lists/*
36
37# Configure ALSA to use PulseAudio (the bridge with the host)
38RUN echo 'pcm.!default { type pulse }' > /etc/asound.conf && \
39 echo 'ctl.!default { type pulse }' >> /etc/asound.conf
40
41# 2. Setup workspace
42WORKDIR /home/ubuntu/home_robot
43
44# Cache optimization: download voice models BEFORE copying the entire code
45# This way, the 150MB download is not repeated if a Python line is changed
46COPY scripts/install_voice_models.sh scripts/
47RUN chmod +x scripts/install_voice_models.sh && \
48 ./scripts/install_voice_models.sh
49
50# 3. Install Python dependencies (before code for better stability)
51RUN pip3 install --break-system-packages \
52 vosk \
53 sounddevice \
54 pyserial \
55 pyyaml
56
57# 4. Copy the rest of the project
58COPY . .
59
60# Ensure ubuntu user has audio and serial permissions
61RUN usermod -aG audio,dialout ubuntu
62
63# Ensure ubuntu user owns the workspace folder
64RUN chown -R ubuntu:ubuntu /home/ubuntu/home_robot
65
66# 5. Build ROS 2 Workspace
67RUN . /opt/ros/jazzy/setup.sh && \
68 cd ros2_ws && \
69 rm -rf build install log && \
70 colcon build --symlink-install
71
72# 6. Environment configuration
73ENV PROJECT_ROOT=/home/ubuntu/home_robot
74ENV RMW_IMPLEMENTATION=rmw_cyclonedds_cpp
75
76# Source setups automatically in ubuntu user's .bashrc
77RUN echo "source /opt/ros/jazzy/setup.bash" >> /home/ubuntu/.bashrc && \
78 echo "source /home/ubuntu/home_robot/ros2_ws/install/setup.bash" >> /home/ubuntu/.bashrc && \
79 echo "export PROJECT_ROOT=/home/ubuntu/home_robot" >> /home/ubuntu/.bashrc
80
81# Let the base image's ENTRYPOINT handle permissions and startup
82# We don't force 'ubuntu' user here to avoid permission errors in logs
